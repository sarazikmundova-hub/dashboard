<!DOCTYPE html>

<html lang="cs">

<head>

   <meta charset="UTF-8">

   <meta name="viewport" content="width=device-width, initial-scale=1.0">

   <title>Social Media Dashboard - Live</title>

   <script src="https://cdn.tailwindcss.com"></script>

   <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

   <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

   <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

   <script src="https://unpkg.com/recharts/umd/Recharts.js"></script>

   <script src="https://unpkg.com/lucide@latest"></script>

   <script src="https://cdnjs.cloudflare.com/ajax/libs/PapaParse/5.3.2/papaparse.min.js"></script>

   <style>

       @import url('https://fonts.googleapis.com/css2?family=Inter:wght@400;600;700;800&display=swap');

       body { font-family: 'Inter', sans-serif; }

       .loading-shimmer {

           background: linear-gradient(90deg, #f0f0f0 25%, #e0e0e0 50%, #f0f0f0 75%);

           background-size: 200% 100%;

           animation: shimmer 1.5s infinite;

       }

       @keyframes shimmer {

           0% { background-position: 200% 0; }

           100% { background-position: -200% 0; }

       }

   </style>

</head>

<body class="bg-slate-50 text-slate-900">

   <div id="root"></div>

   <script type="text/babel">

       const { useState, useMemo, useEffect } = React;

       const { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer, AreaChart, Area, BarChart, Bar } = Recharts;

       // --- DŮLEŽITÉ: ZDE NAHRAĎTE TENTO ODKAZ VAŠÍM ODKAZEM Z GOOGLE SHEETS (Soubor -> Publikovat na web -> CSV) ---

       const GOOGLE_SHEET_CSV_URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vS-V4-q3I7u6HjZt_oK_0YF-O_YyY_your_published_id/pub?gid=32000210&output=csv";

       const Icon = ({ name, size = 20, color = "currentColor" }) => {

           useEffect(() => { if (window.lucide) window.lucide.createIcons(); }, [name]);

           return <i data-lucide={name} style={{ width: size, height: size, color }}></i>;

       };

       const StatCard = ({ title, value, prevValue, iconName, color, suffix = "", loading }) => {

           const diff = (prevValue && value) ? ((value - prevValue) / prevValue) * 100 : 0;

           if (loading) return <div className="h-32 rounded-xl loading-shimmer border border-slate-100"></div>;

           return (

               <div className="bg-white rounded-xl shadow-sm border border-slate-100 p-6 flex flex-col justify-between transition-all hover:shadow-md">

                   <div className="flex justify-between items-start mb-4">

                       <div className="p-3 rounded-lg bg-slate-50" style={{ color }}><Icon name={iconName} /></div>

                       {prevValue !== undefined && (

                           <div className={`flex items-center text-xs font-semibold px-2 py-1 rounded-full ${diff >= 0 ? 'bg-emerald-50 text-emerald-600' : 'bg-rose-50 text-rose-600'}`}>

                               {diff >= 0 ? '+' : ''}{diff.toFixed(1)}%

                           </div>

                       )}

                   </div>

                   <div>

                       <p className="text-slate-500 text-sm font-medium">{title}</p>

                       <h3 className="text-2xl font-bold text-slate-800">

                           {typeof value === 'number' ? value.toLocaleString() : (value || 0)}{suffix}

                       </h3>

                   </div>

               </div>

           );

       };

       const App = () => {

           const [data, setData] = useState([]);

           const [loading, setLoading] = useState(true);

           const [error, setError] = useState(null);

           const [selectedRedakce, setSelectedRedakce] = useState('');

           const [selectedPlatform, setSelectedPlatform] = useState('Všechny');

           useEffect(() => {

               const fetchData = async () => {

                   try {

                       Papa.parse(GOOGLE_SHEET_CSV_URL, {

                           download: true,

                           header: true,

                           dynamicTyping: true,

                           complete: (results) => {

                               const formatted = results.data.map(row => ({

                                   date: row['Měsíc'] || row['date'],

                                   redakce: row['Redakce'] || row['redakce'],

                                   platform: row['Platforma'] || row['platform'],

                                   followers: row['Followers'] || row['followers'] || 0,

                                   interactions: row['Interakce'] || row['interactions'] || 0,

                                   views: row['Views'] || row['views'] || 0,

                                   er: row['ER'] || row['er'] || 0

                               })).filter(d => d.redakce);

                               setData(formatted);

                               if (formatted.length > 0) setSelectedRedakce(formatted[0].redakce);

                               setLoading(false);

                           },

                           error: (err) => {

                               setError("Nepodařilo se načíst data. Zkontrolujte 'Publikovat na web' v Google Sheets.");

                               setLoading(false);

                           }

                       });

                   } catch (e) {

                       setError("Chyba při připojení k tabulce.");

                       setLoading(false);

                   }

               };

               fetchData();

           }, []);

           const redakceOptions = useMemo(() => [...new Set(data.map(d => d.redakce))], [data]);

           const platformOptions = useMemo(() => ['Všechny', ...new Set(data.map(d => d.platform))], [data]);

           const filteredData = useMemo(() => {

               let d = data.filter(item => item.redakce === selectedRedakce);

               if (selectedPlatform !== 'Všechny') d = d.filter(item => item.platform === selectedPlatform);

             

               if (selectedPlatform === 'Všechny') {

                   const grouped = d.reduce((acc, curr) => {

                       if (!acc[curr.date]) {

                           acc[curr.date] = { ...curr, platform: 'Všechny' };

                       } else {

                           acc[curr.date].followers += curr.followers;

                           acc[curr.date].interactions += curr.interactions;

                           acc[curr.date].views += curr.views;

                           acc[curr.date].er = (acc[curr.date].er + curr.er) / 2;

                       }

                       return acc;

                   }, {});

                   return Object.values(grouped).sort((a, b) => a.date.localeCompare(b.date));

               }

               return d.sort((a, b) => a.date.localeCompare(b.date));

           }, [data, selectedRedakce, selectedPlatform]);

           const current = filteredData[filteredData.length - 1] || {};

           const previous = filteredData[filteredData.length - 2] || {};

           if (error) return (

               <div className="flex items-center justify-center min-h-screen p-4 text-center">

                   <div className="max-w-md p-8 bg-white rounded-2xl shadow-xl border border-rose-100">

                       <div className="text-rose-500 mb-4"><Icon name="alert-circle" size={48} className="mx-auto" /></div>

                       <h2 className="text-xl font-bold mb-2">Chyba připojení</h2>

                       <p className="text-slate-600 mb-6">{error}</p>

                       <button onClick={() => window.location.reload()} className="bg-slate-900 text-white px-6 py-2 rounded-lg font-semibold hover:bg-slate-800 transition-colors">Zkusit znovu</button>

                   </div>

               </div>

           );

           return (

               <div className="min-h-screen p-4 md:p-8">

                   <div className="max-w-7xl mx-auto mb-8 flex flex-col md:flex-row md:items-center justify-between gap-4">

                       <div>

                           <h1 className="text-3xl font-extrabold text-slate-900 tracking-tight flex items-center gap-3">

                               Live Dashboard

                               <span className="flex h-3 w-3 rounded-full bg-emerald-500 animate-pulse"></span>

                           </h1>

                           <p className="text-slate-500 mt-1">Automaticky aktualizovaná data z Google Sheets</p>

                       </div>

                     

                       <div className="flex flex-wrap items-center gap-3">

                           <select

                               className="bg-white border border-slate-200 rounded-lg px-4 py-2 shadow-sm font-semibold outline-none focus:ring-2 focus:ring-blue-500"

                               value={selectedRedakce}

                               onChange={(e) => setSelectedRedakce(e.target.value)}

                               disabled={loading}

                           >

                               {redakceOptions.map(opt => <option key={opt} value={opt}>{opt}</option>)}

                           </select>

                           <select

                               className="bg-white border border-slate-200 rounded-lg px-4 py-2 shadow-sm font-semibold outline-none focus:ring-2 focus:ring-blue-500"

                               value={selectedPlatform}

                               onChange={(e) => setSelectedPlatform(e.target.value)}

                               disabled={loading}

                           >

                               {platformOptions.map(opt => <option key={opt} value={opt}>{opt}</option>)}

                           </select>

                       </div>

                   </div>

                   <div className="max-w-7xl mx-auto space-y-6">

                       <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">

                           <StatCard title="Followers" value={current.followers} prevValue={previous.followers} iconName="users" color="#3b82f6" loading={loading} />

                           <StatCard title="Interakce" value={current.interactions} prevValue={previous.interactions} iconName="share-2" color="#10b981" loading={loading} />

                           <StatCard title="Zobrazení" value={current.views} prevValue={previous.views} iconName="eye" color="#f59e0b" loading={loading} />

                           <StatCard title="ER (%)" value={current.er} prevValue={previous.er} iconName="award" color="#8b5cf6" suffix="%" loading={loading} />

                       </div>

                       {!loading && (

                           <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">

                               <div className="bg-white p-6 rounded-xl border border-slate-200 shadow-sm h-80">

                                   <h4 className="font-bold mb-4">Trend růstu komunity</h4>

                                   <ResponsiveContainer width="100%" height="100%">

                                       <AreaChart data={filteredData}>

                                           <CartesianGrid strokeDasharray="3 3" vertical={false} stroke="#f1f5f9" />

                                           <XAxis dataKey="date" tick={{fontSize: 11}} />

                                           <YAxis tick={{fontSize: 11}} />

                                           <Tooltip />

                                           <Area type="monotone" dataKey="followers" stroke="#3b82f6" fill="#3b82f6" fillOpacity={0.1} strokeWidth={3} />

                                       </AreaChart>

                                   </ResponsiveContainer>

                               </div>

                               <div className="bg-white p-6 rounded-xl border border-slate-200 shadow-sm h-80">

                                   <h4 className="font-bold mb-4">Měsíční aktivita (Interakce)</h4>

                                   <ResponsiveContainer width="100%" height="100%">

                                       <BarChart data={filteredData}>

                                           <CartesianGrid strokeDasharray="3 3" vertical={false} stroke="#f1f5f9" />

                                           <XAxis dataKey="date" tick={{fontSize: 11}} />

                                           <YAxis tick={{fontSize: 11}} />

                                           <Tooltip />

                                           <Bar dataKey="interactions" fill="#10b981" radius={[4, 4, 0, 0]} />

                                       </BarChart>

                                   </ResponsiveContainer>

                               </div>

                           </div>

                       )}

                   </div>

               </div>

           );

       };

       const root = ReactDOM.createRoot(document.getElementById('root'));

       root.render(<App />);

   </script>

</body>

</html>
